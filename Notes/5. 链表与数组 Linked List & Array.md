## Linked List的理解
### A linked List class with print function
```py
class ListNode:
    def __init__(self, data):
        self.value = data
        self.next = None
       
class LinkedList:
    def __init__(self, head = None):
        self.head = head

def printLinkedList(self):
    node = self.head
    while node:
        print(node.value)
        node = node.next
        if node:
            print('->')
        
ll = LinkedList()
node1 = ListNode(1)
node2 = ListNode(2)
node3 = ListNode(3)

ll.head = node1
node1.next = node2
node2.next = node3

printLinkedList(ll)
node1 = node2
printLinkedList(ll)

## 结果都是 “1 -> 2 -> 3"
```
